              function doSort(field, direction)
{
	var theForm = document.forms["webOffersQuickFilterForm"];
	theForm.sortFieldName.value=field;
	theForm.sortDirection.value=direction;
	theForm.page.value=0;
	theForm.submit();
}

function submitForm()
{
	var theForm = document.forms["webOffersQuickFilterForm"];
	theForm.submit();
}

function setupQuickFilter()
{
	var theForm = document.forms["webOffersQuickFilterForm"];
	var selectedNum;
	selectedNum = theForm.offerStatusAndReply.selectedIndex;
	if (selectedNum > 7)
	{
		theForm.offerResponse.value = theForm.offerStatusAndReply.options[selectedNum].value;
		theForm.offerStatus.value = "";
	}
	else if (selectedNum > 1 && selectedNum < 7)
	{
		theForm.offerStatus.value = theForm.offerStatusAndReply.options[selectedNum].value;
		theForm.offerResponse.value = "";
	}
	else
	{
		theForm.offerStatus.value = "";
		theForm.offerResponse.value = "";
	}
}

function respondToSelected(responseType)
{
	var theForm = document.forms["webOffersForm"];
	theForm.shipmentIdList.value = checkedToList(theForm.shipmentId) ;
	if (theForm.shipmentIdList.value == "")
	{
		// TODO : Move it to ResourceBundle
		alert('You must select at least one shipment to respond to.');
	}
	else
	{
		theForm.responseType.value = responseType;
		theForm.submit();
	}
}

function checkSizeAgainstMax(textArea, str, maxLen)
{
    if (str.length > maxLen)
	{
         textArea.value = textArea.value.substring(0, maxLen);
    }
}
