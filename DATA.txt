            function AddTotal() {

                var grandTotal = userChargeParams.totalReadonlyCosts;

            function ConvertCultureFormattedDecimalToJsDecimal(formattedStr) {
                if (formattedStr == null || formattedStr == '') return 0;
                var value = formattedStr;
                
                if (userChargeParams.NumberGroupSeparator) { //remove group separater
                    value = value.replace(userChargeParams.NumberGroupSeparator, '');
                }
                if (userChargeParams.NumberDecimalSeparator) { //ensure decimal separater is a period for js conversion
                    value = value.replace(userChargeParams.NumberDecimalSeparator, '.');
                }

                return parseFloat(value);
            }
            
            function FormatJsDecimalToCultureFormattedDecimal(decimalValue) {
                var formattedValue = isNaN(decimalValue) ? '0.00' : decimalValue.toFixed(2);

                var valueSections = formattedValue.split('.');
               
                if (userChargeParams.NumberGroupSeparator) { //add culture specific group separators 
                    valueSections[0] = addGroupSeperators(valueSections[0]);
                }

                if (userChargeParams.NumberDecimalSeparator) { //convert to culter decimal separator
                    formattedValue = valueSections.join(userChargeParams.NumberDecimalSeparator);
                } else {
                    formattedValue = valueSections.join('.');
                }

                return formattedValue;

            function addGroupSeperators(str) {
                var amount = new String(str);
                amount = amount.split("").reverse();

                var output = "";
                for (var i = 0; i <= amount.length - 1; i++) {
                    output = amount[i] + output;
